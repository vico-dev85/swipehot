/* ✅ Safer background preloader with cleanup + error handling */
async function preloadInBackground(src) {
  if (!src) return false;

  const backgroundIframe = getInactiveIframe();

  return new Promise((resolve) => {
    let settled = false;

    const cleanup = () => {
      backgroundIframe.removeEventListener('load', onLoad);
      backgroundIframe.removeEventListener('error', onError);
      clearTimeout(timeout);
    };

    const onLoad = () => {
      if (settled) return;
      settled = true;
      cleanup();
      window.__wetroulette?.applyCrop?.();
      console.log('✅ Background iframe loaded and ready');
      resolve(true);
    };

    const onError = () => {
      if (settled) return;
      settled = true;
      cleanup();
      console.log('❌ Background iframe error while loading');
      resolve(false);
    };

    const timeout = setTimeout(() => {
      if (settled) return;
      settled = true;
      cleanup();
      console.log('⏰ Background preload timeout');
      resolve(false);
    }, 8000);

    backgroundIframe.addEventListener('load', onLoad);
    backgroundIframe.addEventListener('error', onError);
    backgroundIframe.src = src;
    console.log('🔄 Loading in background iframe:', src);
  });
}

function startPreloadTimer() {
  // Clear any existing timer
  clearTimeout(preloadTimer);
  preloadedSrc = null;
  
  // Wait 12 seconds before preloading (indicates user engagement)
  preloadTimer = setTimeout(async () => {
    try {
      console.log('🔄 Fetching next match URL...');
      preloadStartTime = performance.now();
      
      const fetchedSrc = await fetchPreferredNext();

      
      if (fetchedSrc) {
        const fetchTime = Math.round(performance.now() - preloadStartTime);
        console.log(`📡 URL fetched in ${fetchTime}ms, starting background load...`);
        
        // Start loading in background iframe
        const loadSuccess = await preloadInBackground(fetchedSrc);
        
        if (loadSuccess) {
          preloadedSrc = fetchedSrc;
          console.log('🚀 Seamless transition ready - next click will be instant!');
        } else {
          console.log('❌ Background loading failed - will use overlay method');
        }
      } else {
        console.log('⚠️ No valid URL returned from server');
      }
    } catch(e) {
      console.log('❌ Preload failed:', e.message);
      preloadedSrc = null;
    }
  }, 100);
}

function clearPreloadTimer() {
  clearTimeout(preloadTimer);
  preloadTimer = null;
}

  const pool = genderToPool(genderKey);

  // Update label and persist
  setFilterLabel(pool);
  currentPool = pool;
  try { localStorage.setItem('wr_pool', currentPool); } catch {}

  // Close dropdown
  if (isDropdownOpen) toggleGenderDropdown();

  // Optional event
  window.dispatchEvent(new CustomEvent('wr:poolChanged', { detail: { pool: currentPool }}));

  // Reset any old preload and show overlay while we fetch from the new pool
  clearPreloadTimer();
  preloadedSrc = null;
  showTransitionOverlay();

  // Warm and swap smoothly from the new pool
  fetchPreferredNext()
    .then(src => {
      if (!src) throw new Error('no src');
      return preloadInBackground(src).then(() => src);
    })
    .then(src => {
      // Inactive iframe is ready → swap, hide overlay, start next preloader
      swapIframes();

    })
    .catch(() => {
      // Fallback
      window.__wetroulette?.setLiveSrc(STANDBY_LIVE_SRC);
      hideTransitionOverlay();
      startPreloadTimer();
    });
}

function htmlToSrc(html){
  if (!html) return null;
  const m = String(html).match(/src=["']([^"']+)["']/i);
  return m ? m[1] : null;
}
function buildEmbedFromUsername(u){
  if (!u) return null;
  const uname = encodeURIComponent(u);
  return `https://www.xcam.vip/embed/${uname}/?campaign=roGHG&disable_sound=1&embed_video_only=1&join_overlay=1&mobileRedirect=auto&room=${uname}&tour=9oGW&track=wetroulette`;
}
function pickEmbed(o){
  if (!o) return null;

  // --- helpers (use your existing htmlToSrc/buildEmbedFromUsername) ---
  const isListItem = (x) =>
    !!(x && (x.type === 'list' || x.kind === 'list' || x.is_list === true || x.list_embed_src));

  function firstUrlFrom(x){
    if (!x) return null;

    // Wrapped payloads
    if (x.result) { const s = pickEmbed(x.result); if (s) return s; }
    if (x.data)   { const s = pickEmbed(x.data);   if (s) return s; }

    // Explicit list embed
    if (typeof x.list_embed_src === 'string') return x.list_embed_src;

    // Known direct URL fields
    if (typeof x.embed_src === 'string')   return x.embed_src;
    if (typeof x._player_src === 'string') return x._player_src;
    if (typeof x.url === 'string' && /^https?:\/\//i.test(x.url)) return x.url;

    // iframe HTML
    if (typeof x.iframe_embed === 'string') {
      const s = htmlToSrc(x.iframe_embed); if (s) return s;
    }
    if (typeof x.iframe_embed_revshare === 'string') {
      const s = htmlToSrc(x.iframe_embed_revshare); if (s) return s;
    }

    // Username → build embed
    const u = x.username || x.room || x.name || x.model;
    if (typeof u === 'string' && u.trim()) return buildEmbedFromUsername(u.trim());

    // Plain string (URL / iframe / username)
    if (typeof x === 'string'){
      const fromHtml = htmlToSrc(x); if (fromHtml) return fromHtml;
      if (/^https?:\/\//i.test(x)) return x;
      return buildEmbedFromUsername(x.trim());
    }
    return null;
  }

  // Arrays: prefer list items first, then anything else
  if (Array.isArray(o)){
    const listFirst = o.find(isListItem);
    if (listFirst){
      const s = firstUrlFrom(listFirst);
      if (s) return s;
    }
    for (const it of o){
      const s = firstUrlFrom(it);
      if (s) return s;
    }
    return null;
  }

  // Single object: if it's a list, prefer its list embed; else normal resolution
  if (isListItem(o)) {
    const s = firstUrlFrom(o);
    if (s) return s;
  }
  return firstUrlFrom(o);
}

async function loadNext(){
  // Prevent multiple simultaneous transitions
  if (isTransitioning) {
    console.log('🔒 Transition already in progress, ignoring click');
    return;
  }
  
  // Clear any active preload timer since we're switching now
  clearPreloadTimer();
  
  // Check if we have a preloaded match ready in background iframe
  if (preloadedSrc) {
    console.log('⚡ SEAMLESS TRANSITION - No overlay needed!');
    isTransitioning = true;
    
    // Instant swap - no overlay, no waiting!
    swapIframes();
    
    // Reset and start preloading next match
    preloadedSrc = null;
    setTimeout(() => {
      onMatchStart?.();
      startPreloadTimer();
      isTransitioning = false;
    }, 100);
    
    return;
  }
  
  // Fallback: No preload available, use overlay method
  console.log('⏳ No seamless transition ready - using overlay');
  isTransitioning = true;
  showTransitionOverlay();
  
  const msg = document.getElementById('connectingMessage');
  try{
    if (msg) { msg.style.display='block'; msg.textContent='Finding a live room...'; }
    const embed = await fetchPreferredNext();
    if (embed){
      window.__wetroulette?.setLiveSrc(embed);
    } else {
      // No source at all — keep overlay a hair longer but don’t flash a random room
      console.warn('No embed available from hot/legacy/prime.');
    }
    
  }catch(e){
    console.warn('loadNext error', e);
    window.__wetroulette?.setLiveSrc(STANDBY_LIVE_SRC);
  }finally{
    if (msg) msg.style.display='none';
    // Regular 2.5s timeout for on-demand loading
    setTimeout(() => {
      hideTransitionOverlay();
      // Start preloading for next time
      startPreloadTimer();
      isTransitioning = false;
    }, 3000);
  }
}

const RES_EXPIRES_KEY = 'vip_res_expires_at';  // epoch ms when hold ends

function setReservationHold(minutes = 30) {
  const expiresAt = Date.now() + minutes * 60 * 1000;
  try { localStorage.setItem(RES_EXPIRES_KEY, String(expiresAt)); } catch(_) {}
  return expiresAt;
}
function getReservationExpiresAt() {
  const raw = localStorage.getItem(RES_EXPIRES_KEY);
  const t = raw ? parseInt(raw, 10) : 0;
  return Number.isFinite(t) ? t : 0;
}
function clearReservationHold() {
  try { localStorage.removeItem(RES_EXPIRES_KEY); } catch(_) {}
}
function isReservationActive() {
  const t = getReservationExpiresAt();
  return t > Date.now();
}
// ==== Device + provider detection ====
function isMobileUA(){
  return /iphone|ipad|ipod|android|mobile/i.test(navigator.userAgent || '');
}
function getEmailDomain(addr){
  const m = String(addr || '').toLowerCase().match(/@([^> ]+)/);
  return m ? m[1] : '';
}
function providerFromDomain(domain){
  if (!domain) return 'generic';
  if (/(gmail\.com|googlemail\.com)$/.test(domain)) return 'gmail';
  if (/(hotmail\.com|outlook\.com|live\.com|msn\.com)$/.test(domain)) return 'outlook';
  if (/(yahoo\.(com|co\.\w+))$/.test(domain)) return 'yahoo';
  if (/icloud\.com$/.test(domain)) return 'icloud';
  if (/(proton\.me|protonmail\.com)$/.test(domain)) return 'proton';
  if (/aol\.com$/.test(domain)) return 'aol';
  return 'generic';
}

// App schemes (best-effort; harmless if not installed)
function appSchemeFor(provider, toEmail){
  switch(provider){
    case 'gmail':   return `googlegmail:///co?to=${encodeURIComponent(toEmail)}`;
    case 'outlook': return `ms-outlook://compose?to=${encodeURIComponent(toEmail)}`; // iOS Outlook
    case 'yahoo':   return `ymail://mail/compose?to=${encodeURIComponent(toEmail)}`;
    case 'aol':     return `aolmail://mail/compose?to=${encodeURIComponent(toEmail)}`;
    default:        return ''; // icloud/proton/generic: no reliable public compose scheme
  }
}

// Webmail home or search page
function webmailUrlFor(provider){
  switch(provider){
    case 'gmail':   return 'https://mail.google.com/';
    case 'outlook': return 'https://outlook.live.com/mail/';
    case 'yahoo':   return 'https://mail.yahoo.com/';
    case 'icloud':  return 'https://www.icloud.com/mail/';
    case 'proton':  return 'https://mail.proton.me/';
    case 'aol':     return 'https://mail.aol.com/';
    default:        return ''; // unknown → no direct webmail
  }
}

// Try app scheme → then webmail → finally mailto
function openMyMailbox(){
  const to = (localStorage.getItem('vip_email') || '').trim();
  const provider = providerFromDomain(getEmailDomain(to));
  const isMobile = isMobileUA();

  // 1) On mobile, attempt app scheme (non-blocking + graceful timeout)
  if (isMobile){
    const scheme = appSchemeFor(provider, to);
    if (scheme){
      const t0 = Date.now();
      const iframe = document.createElement('iframe');
      iframe.style.display = 'none';
      iframe.src = scheme;
      document.body.appendChild(iframe);

      // After ~800ms, if the tab is still here, fall back to webmail
      setTimeout(() => {
        document.body.removeChild(iframe);
        // If user didn’t switch apps, continue to webmail
        if (Date.now() - t0 < 2200){
          const url = webmailUrlFor(provider);
          if (url) { window.open(url, '_blank'); }
          else { window.location.href = `mailto:${encodeURIComponent(to)}`; }
        }
      }, 800);
      return;
    }
  }

  // 2) Webmail in a new tab (desktop + mobile fallback)
  const url = webmailUrlFor(provider);
  if (url){ window.open(url, '_blank'); return; }

  // 3) Last resort: mailto
  if (to) window.location.href = `mailto:${encodeURIComponent(to)}`;
}

async function handleVipSignupReal(e){
  const btn  = e?.currentTarget || document.getElementById('vipEmailSubmit');
  const form = btn?.closest('.vip-email-form');
  if (!form) return;

  const input = form.querySelector('input[type="email"]');
  const msgEl = form.querySelector('#vipEmailMsg');
  const hp    = form.querySelector('#vipWebsitePot');

  const email = (input?.value || '').trim();
  const showMsg = t => { if (msgEl) msgEl.textContent = t; };

  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) {
    showMsg('Please enter a valid email.');
    return;
  }

  if (btn) btn.disabled = true;
  showMsg('Sending…');

  try {
    await sendVipInvite(email, hp?.value || '', Date.now());  // ← use shim with ts
    showMsg('Invite sent! Check your inbox (Spam/Promotions too).');

    // persist + flip
    try { localStorage.setItem('vip_email', email); } catch(_){}
    if (!window.__wetroulette?.isReservationActive?.()) {
      window.__wetroulette?.setReservationHold?.(30);
    }
    if (typeof markInviteSent === 'function') markInviteSent(true);
    else { try { localStorage.setItem('vip_invite_sent','1'); } catch(_){} }

    // hard flip: clear action slot then render countdown
    const slot = document.getElementById('vipActionSlot');
    if (slot) slot.innerHTML = '';
    window.__wetroulette?.renderActiveReservationCountdown?.();
  } catch (err) {
    console.error(err);
    showMsg('Could not send email. Please try again in a minute.');
  } finally {
    if (btn) btn.disabled = false;
  }
}
